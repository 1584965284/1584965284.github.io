  1. 然后dispatch函数还要调用listeners里所有订阅此store的回调函数

以下是全部代码

const createStore=(reducer,initialState={})=>{
  let currentReducer=reducer;
  let state=initialState
  const listeners=[];
  const subscribe=(listener)=>{
    listeners.push(listener);
    return ()=>{ //unsubscribe取消订阅
      const index=listeners.indexOf(listener);
      listeners.splice(index,1);
    }
  }
  const getState=()=>state;
  const dispatch=(action)=>{
    state=currentReducer(state,action);
    listeners.forEach(listener=>listener());
    return action;
  }
  dispatch({type:'@@REDUX/INIT'});
  return {//store
    subscribe,
    dispatch,
    getState,
  }
}

# combineReducers

创建多个reducer，用combineReducers(reducers)

返回一个总的reducer 遍历传入的reducers对象并执行对应的reducer 返回nextState

const combineReducers=(reducers)=>{
  return function (state={},action){ //reducer(state,action)
    let nextState={};
    const reducerKeys=Object.keys(reducers)
    for(let i=0;i<reducerKeys.length;++i){
      const key=reducerKeys[i];
      nextState[key]=reducers[key](state[key],action);
    }
    return nextState;
  }
}//action.type一样的话所有reducer都会执行 可以用来reset
