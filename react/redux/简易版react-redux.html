# connect

export function connect(mapStateToProps,mapDispatchToProps){
  return function (OldComponent){ //返回一个高阶组件
    return function (props){  //返回的那个组件
      const {store}=useContext(ReactReduxContext)
      const {getState,dispatch,subscribe}=store;
      //把状态映射为属性
      const stateProps=mapStateToProps(getState());
      //把dispatch映射为属性
      let dispatchProps= useMemo(() => {
        let dispatchProps
        if(typeof mapDispatchToProps==='function'){
          dispatchProps=mapDispatchToProps(dispatch);
        }else if(typeof mapDispatchToProps==='object'){
          // dispatchProps=bindActionCreators(mapDispatchToProps,dispatch)
        }else{
          dispatchProps={dispatch};
        }
        return dispatchProps;
      },[dispatch]);

      const [,foreUpdate]=useReducer(tmpState=>tmpState+1,0);//模拟类组件强制刷新

      useLayoutEffect(()=>{
        return subscribe(foreUpdate)
      },[subscribe]);

      return <OldComponent {...props} {...stateProps} {...dispatchProps}/>
    }
  }
}

# hooks

export const useDispatch=()=>{
  const {store}=useContext(ReactReduxContext);
  return store.dispatch;
}

export const useSelector=(selector)=>{
  const {store}=useContext(ReactReduxContext);
  const {subscribe,getState}=store;
  const [,foreUpdate]=useReducer(x=>x+1,0);

  useLayoutEffect(()=>{
    subscribe(foreUpdate)
  },[subscribe]);

  return selector(getState());
}
